020package com.garygregory.jcommander.converters.time;
021
022import java.time.ZoneOffset;
023
024import com.garygregory.jcommander.converters.AbstractBaseConverter;
025
026/**
027 * Converts a {@link String} into a {@link ZoneOffset}.
028 * <p>
029 * For a description of the format, see {@link ZoneOffset#of(String)}.
030 * </p>
031 * 
032 * <p>
033 * Example:
034 * </p>
035 * 
036 * <pre class="prettyprint">
037 * <code class="language-java">&#64;Parameter(names = { "--zoneOffset" }, converter = ZoneOffsetConverter.class)
038 * private ZoneOffset zoneOffset;</code>
039 * </pre>
040 * <p>
041 * 
042 * @see ZoneOffset
043 * @see ZoneOffset#of(String)
044 * 
045 * @since 1.0.0
046 * @author <a href="mailto:ggregory@garygregory.com">Gary Gregory</a>
047 */
048public class ZoneOffsetConverter extends AbstractBaseConverter<ZoneOffset> {
049
050    /**
051     * Constructs a converter.
052     * 
053     * @param optionName
054     *            The option name, may be null.
055     */
056    public ZoneOffsetConverter(final String optionName) {
057        super(optionName, ZoneOffset.class);
058    }
059
060    @Override
061    protected ZoneOffset convertImpl(final String value) {
062        return ZoneOffset.of(value);
063    }
064
065}
