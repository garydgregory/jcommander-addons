020package com.garygregory.jcommander.converters.sql.mysql;
021
022import java.sql.Types;
023
024import com.garygregory.jcommander.converters.AbstractBaseConverter;
025import com.garygregory.jcommander.converters.sql.TypesConverter;
026
027/**
028 * Converts a {@link String} into a SQL Types {@link Integer}.
029 * <p>
030 * For a description of the format, see {@link Types}.
031 * </p>
032 * 
033 * <p>
034 * Example:
035 * </p>
036 * 
037 * <pre class="prettyprint">
038 * <code class="language-java">&#64;Parameter(names = { "--types" }, converter = TypesConverter.class)
039 * private Types types;</code>
040 * </pre>
041 * <p>
042 * 
043 * @see Types
044 * 
045 * @since 1.0.0
046 * @author <a href="mailto:ggregory@garygregory.com">Gary Gregory</a>
047 */
048public class MysqlTypesConverter extends AbstractBaseConverter<Integer> {
049
050    /**
051     * Constructs a converter.
052     * 
053     * @param optionName
054     *            The option name, may be null.
055     */
056    public MysqlTypesConverter(final String optionName) {
057        super(optionName, Integer.class);
058    }
059
060    @Override
061    protected Integer convertImpl(final String value) {
062        return Integer.valueOf(toType(value));
063    }
064
065    private int toType(final String value) {
066        switch (value) {
067        case "BIGINT":
068            return Types.BIGINT;
069        case "BIGINT UNSIGNED":
070        case "BIGINT_UNSIGNED":
071            return Types.BIGINT;
072        case "BINARY":
073            return Types.BINARY;
074        case "BIT":
075            return Types.BIT;
076        case "BLOB":
077            return Types.LONGVARBINARY;
078        case "BOOLEAN":
079            return Types.BOOLEAN;
080        case "CHAR":
081            return Types.CHAR;
082        case "DATE":
083            return Types.DATE;
084        case "DATETIME":
085            return Types.TIMESTAMP;
086        case "DECIMAL":
087            return Types.DECIMAL;
088        case "DECIMAL UNSIGNED":
089        case "DECIMAL_UNSIGNED":
090            return Types.DECIMAL;
091        case "DOUBLE":
092            return Types.DOUBLE;
093        case "DOUBLE UNSIGNED":
094        case "DOUBLE_UNSIGNED":
095            return Types.DOUBLE;
096        case "ENUM":
097            return Types.CHAR;
098        case "FLOAT":
099            return Types.REAL;
100        case "FLOAT UNSIGNED":
101        case "FLOAT_UNSIGNED":
102            return Types.REAL;
103        case "GEOMETRY":
104            return Types.BINARY;
105        case "INT":
106            return Types.INTEGER;
107        case "INT UNSIGNED":
108        case "INT_UNSIGNED":
109            return Types.INTEGER;
110        case "JSON":
111            return Types.LONGVARCHAR;
112        case "LONGBLOB":
113            return Types.LONGVARBINARY;
114        case "LONGTEXT":
115            return Types.LONGVARCHAR;
116        case "MEDIUMBLOB":
117            return Types.LONGVARBINARY;
118        case "MEDIUMINT UNSIGNED":
119        case "MEDIUMINT_UNSIGNED":
120            return Types.INTEGER;
121        case "MEDIUMINT":
122            return Types.INTEGER;
123        case "MEDIUMTEXT":
124            return Types.LONGVARCHAR;
125        case "NULL":
126            return Types.NULL;
127        case "SET":
128            return Types.CHAR;
129        case "SMALLINT":
130            return Types.SMALLINT;
131        case "SMALLINT UNSIGNED":
132        case "SMALLINT_UNSIGNED":
133            return Types.SMALLINT;
134        case "TEXT":
135            return Types.LONGVARCHAR;
136        case "TIME":
137            return Types.TIME;
138        case "TIMESTAMP":
139            return Types.TIMESTAMP;
140        case "TINYBLOB":
141            return Types.VARBINARY;
142        case "TINYINT":
143            return Types.TINYINT;
144        case "TINYINT UNSIGNED":
145        case "TINYINT_UNSIGNED":
146            return Types.TINYINT;
147        case "TINYTEXT":
148            return Types.VARCHAR;
149        case "VARBINARY":
150            return Types.VARBINARY;
151        case "VARCHAR":
152            return Types.VARCHAR;
153        case "YEAR":
154            return Types.DATE;
155        default:
156            return TypesConverter.toType(value);
157        }
158    }
159
160}
