020package com.garygregory.jcommander.converters.sql;
021
022import java.sql.Types;
023
024import com.garygregory.jcommander.converters.AbstractBaseConverter;
025
026/**
027 * Converts a {@link String} into a SQL Types {@link Integer}.
028 * <p>
029 * For a description of the format, see {@link Types}.
030 * </p>
031 * 
032 * <p>
033 * Example:
034 * </p>
035 * 
036 * <pre class="prettyprint">
037 * <code class="language-java">&#64;Parameter(names = { "--types" }, converter = TypesConverter.class)
038 * private Types types;</code>
039 * </pre>
040 * <p>
041 * 
042 * @see Types
043 * 
044 * @since 1.0.0
045 * @author <a href="mailto:ggregory@garygregory.com">Gary Gregory</a>
046 */
047public class TypesConverter extends AbstractBaseConverter<Integer> {
048
049    /**
050     * Converts a string to a JDBC SQL type int.
051     * 
052     * @param value
053     *            Type name
054     * @return a {@link Types} value
055     */
056    public static int toType(final String value) {
057        switch (value) {
058        case "ARRAY":
059            return Types.ARRAY;
060        case "BIGINT":
061            return Types.BIGINT;
062        case "BINARY":
063            return Types.BINARY;
064        case "BIT":
065            return Types.BIT;
066        case "BLOB":
067            return Types.BLOB;
068        case "BOOLEAN":
069            return Types.BOOLEAN;
070        case "CHAR":
071            return Types.CHAR;
072        case "CLOB":
073            return Types.CLOB;
074        case "DATALINK":
075            return Types.DATALINK;
076        case "DATE":
077            return Types.DATE;
078        case "DECIMAL":
079            return Types.DECIMAL;
080        case "DISTINCT":
081            return Types.DISTINCT;
082        case "DOUBLE":
083            return Types.DOUBLE;
084        case "FLOAT":
085            return Types.FLOAT;
086        case "INTEGER":
087            return Types.INTEGER;
088        case "JAVA_OBJECT":
089            return Types.JAVA_OBJECT;
090        case "LONGNVARCHAR":
091            return Types.LONGNVARCHAR;
092        case "LONGVARBINARY":
093            return Types.LONGVARBINARY;
094        case "LONGVARCHAR":
095            return Types.LONGVARCHAR;
096        case "NCHAR":
097            return Types.NCHAR;
098        case "NCLOB":
099            return Types.NCLOB;
100        case "NULL":
101            return Types.NULL;
102        case "NUMERIC":
103            return Types.NUMERIC;
104        case "NVARCHAR":
105            return Types.NVARCHAR;
106        case "OTHER":
107            return Types.OTHER;
108        case "REAL":
109            return Types.REAL;
110        case "REF":
111            return Types.REF;
112        case "REF_CURSOR":
113            return Types.REF_CURSOR;
114        case "ROWID":
115            return Types.ROWID;
116        case "SMALLINT":
117            return Types.SMALLINT;
118        case "SQLXML":
119            return Types.SQLXML;
120        case "STRUCT":
121            return Types.STRUCT;
122        case "TIME":
123            return Types.TIME;
124        case "TIME_WITH_TIMEZONE":
125            return Types.TIME_WITH_TIMEZONE;
126        case "TIMESTAMP":
127            return Types.TIMESTAMP;
128        case "TIMESTAMP_WITH_TIMEZONE":
129            return Types.TIMESTAMP_WITH_TIMEZONE;
130        case "TINYINT":
131            return Types.TINYINT;
132        case "VARBINARY":
133            return Types.VARBINARY;
134        case "VARCHAR":
135            return Types.VARCHAR;
136        default:
137            throw new IllegalArgumentException("Unknown JDBC SQL type '" + value + "'");
138        }
139    }
140
141    /**
142     * Constructs a converter.
143     * 
144     * @param optionName
145     *            The option name, may be null.
146     */
147    public TypesConverter(final String optionName) {
148        super(optionName, Integer.class);
149    }
150
151    @Override
152    protected Integer convertImpl(final String value) {
153        return Integer.valueOf(toType(value));
154    }
155
156}
